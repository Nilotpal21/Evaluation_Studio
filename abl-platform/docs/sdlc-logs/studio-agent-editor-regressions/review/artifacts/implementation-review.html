  45   try {
  46     const parsed = JSON.parse(manifestContent) as Record<string, unknown>;

  48 
  49     const rewriteRecords = (records: unknown) => {
  50       if (!records || typeof records !== 'object' || Array.isArray(records)) {
  51         return;
  52       }
  53 
  54       for (const record of Object.values(records as Record<string, unknown>)) {
  55         if (!record || typeof record !== 'object' || Array.isArray(record)) {

  60         if (typeof currentPath !== 'string') {
  61           continue;
  62         }
  63 
  64         const rewrittenPath = normalizedPathMap.get(currentPath);
  65         if (rewrittenPath && rewrittenPath !== currentPath) {
  66           (record as { path: string }).path = rewrittenPath;
  67           changed = true;
  68         }
  69       }
  70     };
  71 
  72     rewriteRecords(parsed.agents);
  73     rewriteRecords(parsed.tools);
  74     rewriteRecords(parsed.behavior_profiles);
  75 
  76     return changed ? JSON.stringify(parsed, null, 2) : null;
